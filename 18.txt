create database pr23;
use pr23;
DROP TABLE IF EXISTS Salary_Difference;
DROP TABLE IF EXISTS CUSTOMERS;

CREATE TABLE CUSTOMERS (
    ID INT PRIMARY KEY,
    NAME VARCHAR(50),
    SALARY DECIMAL(10,2)
);

CREATE TABLE Salary_Difference (
    ID INT AUTO_INCREMENT PRIMARY KEY,
    CUSTOMER_ID INT,
    ACTION_TYPE VARCHAR(20),
    OLD_SALARY DECIMAL(10,2),
    NEW_SALARY DECIMAL(10,2),
    DIFFERENCE DECIMAL(10,2)
);

DELIMITER $$

CREATE TRIGGER customers_insert
AFTER INSERT ON CUSTOMERS
FOR EACH ROW
BEGIN
    INSERT INTO Salary_Difference
    (CUSTOMER_ID, ACTION_TYPE, OLD_SALARY, NEW_SALARY, DIFFERENCE)
    VALUES
    (NEW.ID, 'INSERT', 0, NEW.SALARY, NEW.SALARY);
END$$

DELIMITER $$
CREATE TRIGGER customers_update
AFTER UPDATE ON CUSTOMERS
FOR EACH ROW
BEGIN
    INSERT INTO Salary_Difference
    (CUSTOMER_ID, ACTION_TYPE, OLD_SALARY, NEW_SALARY, DIFFERENCE)
    VALUES
    (NEW.ID, 'UPDATE', OLD.SALARY, NEW.SALARY, NEW.SALARY - OLD.SALARY);
END$$

DELIMITER $$
CREATE TRIGGER customers_delete
BEFORE DELETE ON CUSTOMERS
FOR EACH ROW
BEGIN
    INSERT INTO Salary_Difference
    (CUSTOMER_ID, ACTION_TYPE, OLD_SALARY, NEW_SALARY, DIFFERENCE)
    VALUES
    (OLD.ID, 'DELETE', OLD.SALARY, 0, 0 - OLD.SALARY);
END$$

DELIMITER ;

INSERT INTO CUSTOMERS VALUES (1, 'Pranav', 30000);
UPDATE CUSTOMERS SET SALARY = 35000 WHERE ID = 1;
DELETE FROM CUSTOMERS WHERE ID = 1;

SELECT * FROM Salary_Difference;
