CREATE PROCEDURE libraryfine(
    IN rno INT,
    IN bname VARCHAR(50)
)
BEGIN

    DECLARE d1 DATE;
    DECLARE nodays INT;
    DECLARE fineamt INT DEFAULT 0;

    SELECT DateofIssue
    INTO d1
    FROM Borrower
    WHERE Rollin = rno
    AND NameofBook = bname;

    SET nodays = DATEDIFF(CURDATE(), d1);

    IF nodays BETWEEN 15 AND 30 THEN
        SET fineamt = nodays * 5;

    ELSEIF nodays > 30 THEN
        SET fineamt = (30 * 5) + ((nodays - 30) * 50);

    END IF;

    IF fineamt > 0 THEN
        INSERT INTO Fine
        VALUES(rno, CURDATE(), fineamt);
    END IF;

    UPDATE Borrower
    SET Status = 'R'
    WHERE Rollin = rno
    AND NameofBook = bname;

END //

DELIMITER ;

SET SQL_SAFE_UPDATES = 0;
