DELIMITER //

CREATE PROCEDURE MergeData()
BEGIN

    DECLARE done INT DEFAULT 0;
    DECLARE rno INT;
    DECLARE rname VARCHAR(50);
    DECLARE cnt INT;

    DECLARE cur CURSOR FOR
    SELECT RollNo, Name FROM N_RollCall;

    DECLARE CONTINUE HANDLER FOR NOT FOUND
    SET done = 1;

    OPEN cur;

    read_loop: LOOP

        FETCH cur INTO rno, rname;

        IF done = 1 THEN
            LEAVE read_loop;
        END IF;

        SELECT COUNT(*)
        INTO cnt
        FROM O_RollCall
        WHERE RollNo = rno;

        IF cnt = 0 THEN

            INSERT INTO O_RollCall
            VALUES(rno, rname);

        END IF;

    END LOOP;

    CLOSE cur;

END //

DELIMITER ;


CALL MergeData();
